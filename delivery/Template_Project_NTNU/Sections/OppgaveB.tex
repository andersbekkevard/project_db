\section{Relasjonsdatabaseskjema og normalformer}

I denne delen presenterer vi ER-modellen oversatt til relasjonsdatabaseskjema. For hver tabell oppgir vi nøkler, fremmednøkler og en kort begrunnelse for hvorfor tabellen er på BCNF.

\newcommand{\schemaentry}[1]{%
  \subsubsection*{#1}%
  \vspace{-0.25em}%
}

\newenvironment{schemafks}{%
  \begin{itemize}
    \setlength{\itemsep}{0.05em}
    \setlength{\parskip}{0pt}
    \setlength{\topsep}{0pt}
    \setlength{\partopsep}{0pt}
    \setlength{\parsep}{0pt}
  }{%
  \end{itemize}
}

\schemaentry{bruker(\underline{id}, navn, epost, mobilnr, utestengt\_til)}
\textbf{BCNF:} Alle attributter er kun funksjonelt avhengige av primærnøkkelen id.

\vspace{0.8em}
\schemaentry{medlem(\underline{id}, starttid, gyldig\_til, bruker\_id)}
\textbf{Fremmednøkkel:} bruker\_id $\rightarrow$ bruker(id)\par
\vspace{0.15em}
\textbf{BCNF:} Alle attributter avhenger kun av id.

\vspace{0.8em}
\schemaentry{medlem\_av(\underline{bruker\_id}, \underline{idrettslag\_id}, starttid)}
\textbf{Fremmednøkler:}\par\vspace{-0.55em}
\begin{schemafks}
    \item bruker\_id $\rightarrow$ bruker(id)
    \item idrettslag\_id $\rightarrow$ idrettslag(id)
\end{schemafks}
\vspace{-0.2em}
\textbf{BCNF:} Sammensatt primærnøkkel der starttid avhenger av hele nøkkelen.

\vspace{0.8em}
\schemaentry{idrettslag(\underline{id}, navn)}
\textbf{BCNF:} Alle attributter er funksjonelt avhengige av primærnøkkelen id.

\vspace{0.8em}
\schemaentry{idrettslag\_gruppe(\underline{gruppenavn}, idrettslag\_id)}
\textbf{Fremmednøkkel:} idrettslag\_id $\rightarrow$ idrettslag(id)\par
\vspace{0.15em}
\textbf{BCNF:} idrettslag\_id er avhengig av gruppenavn. Refererer til forutsetningene beskrevet i deloppgave a.

\vspace{0.8em}
\schemaentry{gruppereservasjon(\underline{id}, uke\_nr, gruppenavn, tidsblokk\_starttid, tidsblokk\_ukedag, senter\_id, sal\_nr)}
\textbf{Fremmednøkler:}\par\vspace{-0.55em}
\begin{schemafks}
    \item (gruppenavn, idrettslag\_id) $\rightarrow$ idrettslag\_gruppe(gruppenavn, idrettslag\_id)
    \item (tidsblokk\_starttid, tidsblokk\_ukedag) $\rightarrow$ tidsblokk(starttid, ukedag)
    \item (senter\_id, sal\_nr) $\rightarrow$ sal(senter\_id, nr)
\end{schemafks}
\vspace{-0.2em}
\textbf{BCNF:} Alle attributter avhenger kun av id.

\vspace{0.8em}
\schemaentry{tidsblokk(\underline{starttid}, \underline{ukedag})}
\textbf{BCNF:} Sammensatt primærnøkkel, ingen andre attributter.

\vspace{0.8em}
\schemaentry{gruppetime(\underline{id}, uke\_nr, år, instruktør\_id, aktivitetstype, senter\_id, sal\_nr)}
\textbf{Fremmednøkler:}\par\vspace{-0.55em}
\begin{schemafks}
    \item instruktør\_id $\rightarrow$ instruktør(id)
    \item aktivitetstype $\rightarrow$ aktivitetstype(navn)
    \item (senter\_id, sal\_nr) $\rightarrow$ sal(senter\_id, nr)
\end{schemafks}
\vspace{-0.2em}
\textbf{BCNF:} Alle attributter avhenger kun av id.

\vspace{0.8em}
\schemaentry{senterbesøk(\underline{id}, ankomsttidspunkt, bruker\_id, senter\_id)}
\textbf{Fremmednøkler:}\par\vspace{-0.55em}
\begin{schemafks}
    \item bruker\_id $\rightarrow$ bruker(id)
    \item senter\_id $\rightarrow$ senter(id)
\end{schemafks}
\vspace{-0.2em}
\textbf{BCNF:} Alle attributter avhenger kun av id.

\vspace{0.8em}
\schemaentry{senter(\underline{id}, navn, gateadresse)}
\textbf{BCNF:} Alle attributter avhenger kun av id.

\vspace{0.8em}
\schemaentry{senteråpningstid(\underline{dato}, \underline{senter\_id}, start\_tid, slutt\_tid)}
\textbf{Fremmednøkkel:} senter\_id $\rightarrow$ senter(id)\par
\vspace{0.15em}
\textbf{BCNF:} Start\_tid og slutt\_tid avhenger kun av hele den sammensatte nøkkelen. Dette er den eneste funksjonelle avhengigheten vi har.

\vspace{0.8em}
\schemaentry{senterbemanning(\underline{dato}, \underline{senter\_id}, start\_tid, slutt\_tid)}
\textbf{Fremmednøkkel:} senter\_id $\rightarrow$ senter(id)\par
\vspace{0.15em}
\textbf{BCNF:} Start\_tid og slutt\_tid avhenger kun av hele den sammensatte nøkkelen.

\vspace{0.8em}
\schemaentry{fasilitet(\underline{navn}, ikon)}
\textbf{BCNF:} Ikon avhenger kun av navn som er primærnøkkel.

\vspace{0.8em}
\schemaentry{sal(\underline{senter\_id}, \underline{nr}, kapasitet)}
\textbf{Fremmednøkkel:} senter\_id $\rightarrow$ senter(id)\par
\vspace{0.15em}
\textbf{BCNF:} Kapasitet er avhengig av hele primærnøkkelen.

\vspace{0.8em}
\schemaentry{spinningsal(\underline{senter\_id}, \underline{sal\_nr})}
\textbf{Fremmednøkkel:} (senter\_id, sal\_nr) $\rightarrow$ sal(senter\_id, nr)\par
\vspace{0.15em}
\textbf{BCNF:} Kun primærnøkkel, ingen andre attributter.

\vspace{0.8em}
\schemaentry{løpesal(\underline{senter\_id}, \underline{sal\_nr})}
\textbf{Fremmednøkkel:} (senter\_id, sal\_nr) $\rightarrow$ sal(senter\_id, nr)\par
\vspace{0.15em}
\textbf{BCNF:} Kun primærnøkkel, ingen andre attributter.

\vspace{0.8em}
\schemaentry{flerbrukshall(\underline{senter\_id}, \underline{sal\_nr}, type)}
\textbf{Fremmednøkkel:} (senter\_id, sal\_nr) $\rightarrow$ sal(senter\_id, nr)\par
\vspace{0.15em}
\textbf{BCNF:} Type avhenger av både senter\_id og sal\_nr, derfor på BCNF.

\vspace{0.8em}
\schemaentry{spinningsykkel(\underline{id}, type, har\_bluetooth, senter\_id, sal\_nr)}
\textbf{Fremmednøkkel:} (senter\_id, sal\_nr) $\rightarrow$ spinningsal(senter\_id, sal\_nr)\par
\vspace{0.15em}
\textbf{BCNF:} Alle attributter er kun avhengige av id.

\vspace{0.8em}
\schemaentry{tredemølle(\underline{id}, produsent, maks\_hastighet, maks\_stigning, senter\_id, sal\_nr)}
\textbf{Fremmednøkkel:} (senter\_id, sal\_nr) $\rightarrow$ løpesal(senter\_id, sal\_nr)\par
\vspace{0.15em}
\textbf{BCNF:} Alle attributter avhenger kun av id.

\vspace{0.8em}
\schemaentry{time\_skjer\_i(\underline{gruppetime\_id}, tidsblokk\_starttid, tidsblokk\_ukedag)}
\textbf{Fremmednøkler:}\par\vspace{-0.55em}
\begin{schemafks}
    \item gruppetime\_id $\rightarrow$ gruppetime(id)
    \item (tidsblokk\_starttid, tidsblokk\_ukedag) $\rightarrow$ tidsblokk(starttid, ukedag)
\end{schemafks}
\vspace{-0.2em}
\textbf{BCNF:} Alle attributter avhenger kun av gruppetime\_id.

\vspace{0.8em}
\schemaentry{deltar\_på\_time(\underline{gruppetime\_id}, \underline{bruker\_id}, påmeldt\_tidspunkt, avmeldt\_tidspunkt, prikk\_dato)}
\textbf{Fremmednøkler:}\par\vspace{-0.55em}
\begin{schemafks}
    \item gruppetime\_id $\rightarrow$ gruppetime(id)
    \item bruker\_id $\rightarrow$ bruker(id)
\end{schemafks}
\vspace{-0.2em}
\textbf{BCNF:} Sammensatt nøkkel utgjør venstresiden av alle funksjonelle avhengigheter.

\vspace{0.8em}
\schemaentry{fasiliterer(\underline{senter\_id}, \underline{fasilitetsnavn})}
\textbf{Fremmednøkler:}\par\vspace{-0.55em}
\begin{schemafks}
    \item senter\_id $\rightarrow$ senter(id)
    \item fasilitetsnavn $\rightarrow$ fasilitet(navn)
\end{schemafks}
\vspace{-0.2em}
\textbf{BCNF:} Kun sammensatt primærnøkkel, ingen andre attributter.

\vspace{0.8em}
\schemaentry{instruktør(\underline{id}, fornavn)}
\textbf{BCNF:} Fornavn avhenger av id.

\vspace{0.8em}
\schemaentry{aktivitetstype(\underline{navn}, beskrivelse)}
\textbf{BCNF:} Beskrivelse avhenger kun av navnet som er primærnøkkel.
